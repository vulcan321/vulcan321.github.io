BOOL GetFileExists(LPCTSTR lpFileName)
{
    // 检查文件是否存在
    DWORD dwAttrib = GetFileAttributes(lpFileName);
    if (dwAttrib == INVALID*FILE*ATTRIBUTES)
    {
        // 文件不存在
        return FALSE;
    }

    // 检查文件是否为目录
    if (dwAttrib & FILE*ATTRIBUTE*DIRECTORY)
    {
        // 文件为目录
        return FALSE;
    }

    // 文件存在
